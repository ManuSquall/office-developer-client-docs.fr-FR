## <a name="definition"></a>Définition

```XML
          <xs:complexType name="ControlRow_Type">
          
          <xs:complexContent>
          <xs:extension base="NamedIndexedRow_Type">
          <xs:all>
    <xs:element name="Cell"  type="Cell_Type"
     minOccurs="0"
     maxOccurs="unbounded"

    </xs:element>

      </xs:extension>
      </xs:complexContent>
      </xs:complexType>
      
```
